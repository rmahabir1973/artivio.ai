STEP 4: Backend API Endpoint (Node.js/Express)
Add this endpoint to your Replit backend:

// Handle newsletter signup
app.post('/api/newsletter-signup', async (req, res) => {
  const { email } = req.body;

  if (!email) {
    return res.status(400).json({ error: 'Email required' });
  }

  try {
    // Call Loops.so API
    const response = await fetch('https://app.loops.so/api/v1/contacts/update', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer YOUR_LOOPS_API_KEY` // Replace with actual key
      },
      body: JSON.stringify({
        email: email,
        mailingLists: {
          'cmho6d4k70p4b0i23ae4q7z20': true // Free Trial list
        },
        subscribed: true
      })
    });

    const data = await response.json();

    if (response.ok) {
      res.status(200).json({ message: 'Subscribed successfully!' });
    } else {
      res.status(400).json({ error: data.message || 'Failed to subscribe' });
    }
  } catch (error) {
    console.error('Newsletter signup error:', error);
    res.status(500).json({ error: 'Server error' });
  }
});
Replace YOUR_LOOPS_API_KEY with your actual Loops.so API key from Step 1
